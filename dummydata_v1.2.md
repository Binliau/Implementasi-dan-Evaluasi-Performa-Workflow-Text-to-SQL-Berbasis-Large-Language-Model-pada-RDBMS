### **BAGIAN 3: KALKULASI TABEL STOK YANG AKURAT (FINAL)**

**Penting**: Kueri SQL di bawah ini harus Anda jalankan terhadap database *setelah* semua data dari **BAGIAN 1** dan **BAGIAN 2** di atas telah dimasukkan. Kueri ini akan secara otomatis menghitung stok akhir yang akurat berdasarkan semua transaksi dan membuat pernyataan `INSERT` yang benar untuk tabel `stok`.

```sql
-- Kueri untuk Menghasilkan Perintah INSERT INTO untuk tabel `stok`
-- Jalankan kueri ini setelah semua data transaksi dari BAGIAN 2 dimasukkan.

INSERT INTO `stok` (`barang_id`, `gudang_id`, `jumlah`, `tanggal_kadaluarsa`)
SELECT
    final_stock.barang_id,
    final_stock.gudang_id,
    final_stock.jumlah_stok_akhir,
    (
        SELECT
            r.tanggal_kadaluarsa
        FROM
            restock r
        WHERE
            r.barang_id = final_stock.barang_id AND r.gudang_id = final_stock.gudang_id
        ORDER BY
            r.tanggal DESC
        LIMIT 1
    ) AS tanggal_kadaluarsa
FROM
    (
        SELECT
            b.id AS barang_id,
            g.id AS gudang_id,
            (COALESCE(r.total_restock, 0) - COALESCE(p.total_penjualan, 0) - COALESCE(t_keluar.total_transfer_keluar, 0) + COALESCE(t_masuk.total_transfer_masuk, 0)) AS jumlah_stok_akhir
        FROM
            barang b
        CROSS JOIN
            gudang g
        LEFT JOIN
            (SELECT barang_id, gudang_id, SUM(jumlah) AS total_restock FROM restock GROUP BY barang_id, gudang_id) r ON b.id = r.barang_id AND g.id = r.gudang_id
        LEFT JOIN
            (SELECT barang_id, gudang_id, SUM(jumlah) AS total_penjualan FROM penjualan GROUP BY barang_id, gudang_id) p ON b.id = p.barang_id AND g.id = p.gudang_id
        LEFT JOIN
            (SELECT barang_id, dari_gudang_id, SUM(jumlah) AS total_transfer_keluar FROM transfer GROUP BY barang_id, dari_gudang_id) t_keluar ON b.id = t_keluar.barang_id AND g.id = t_keluar.dari_gudang_id
        LEFT JOIN
            (SELECT barang_id, ke_gudang_id, SUM(jumlah) AS total_transfer_masuk FROM transfer GROUP BY barang_id, ke_gudang_id) t_masuk ON b.id = t_masuk.barang_id AND g.id = t_masuk.ke_gudang_id
    ) AS final_stock
WHERE
    final_stock.jumlah_stok_akhir > 0
ORDER BY
    final_stock.gudang_id, final_stock.barang_id;
```
